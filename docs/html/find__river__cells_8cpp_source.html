409/*
410 * Method to return inflow cell, which is eighter
411 * 1.) the cell itself, if the actual cell has no inflow
412 * 2.) the inflow cell, if there exists only one inflow cell
413 * 3.) the inflow cell with the highest discharge
414 */
415int getInflowCell(int cell, int (*inflow)[8], double *dis){
416 int count, one;
417 count=0; //counts the number of inflow cells
418 one=0; //saves the indice in case of just one inflow cell
419
420 //looks at all neighbouring cell if they are inflow to the actual cell
421 for(int j =0; j<8;j++){
422 if(inflow[cell-1][j]!=0){
423 count++;
424 one=j;
425 }
426 }
427 //decides which of the three possibilities is apparent
428 switch(count){
429 case 0: return cell;
430 case 1: return inflow[cell-1][one];
431 default: return getHiDis(inflow, cell, dis);
432 }
433 //cout<<"getInflowCell done"<<endl;
434 return 0;
435}

461/*
462 * Method to find the hightest discharge of all neighbouring cells
463 */
464int getHiDis(int (*inflow)[8], int cell, double *dis){
465 double max=0.0;
466 int maxcell = cell;
467
468 for(int j =0; j<8; j++) {
469 int neighbour = inflow[cell-1][j];
470 if (neighbour>0 && dis[neighbour-1]>0 && dis[neighbour-1]> max) {
471 max = dis[neighbour-1];
472 maxcell = neighbour;
473 }
474 }
475
476
477 //returns the arc_id of the inflow cell with the highest discharge
478 // oder gleiche Zelle
479 return maxcell;
480}
