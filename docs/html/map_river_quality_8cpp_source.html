427int ClassDefine(double value, short toCalc, short flagTH) {
428
429
430switch(toCalc){
431
432//BOD
433
434case 0: if(flagTH==0){
435 if (value> 0. && value <= 4.) return 1; //no impact on fish
436 else if (value> 4. && value <= 8.) return 2; // low impact on fish
437 else if (value> 8.) return 3; // high impact on fish
438 else return 0;
439
440}else if(flagTH==1){
441 if (value> 0. && value <= 1.) return 1;
442 else if (value> 1. && value <= 2.5) return 2;
443 else if (value> 2.5 && value <= 5.0) return 3;
444 else if (value> 5.0 && value <=10.0) return 4;
445 else if (value>10.0 && value <=50.0) return 5;
446 else if (value>50.0) return 6;
447 else return 0;
448
449
450}else
451 cout<<"flag: "<<flagTH<<endl;
452 cerr << "ERROR: Wrong flag" << endl;
453 cerr << "usage: only (0) impact on fish and (1) general (from SCENES) possible"; exit(1);
454 return NODATA;
455
456//TDS
457
458case 1: if (value> 0. && value <= 450.) return 1;// according to FAO Water Quality for Irrigation guideline none restriction
459 else if (value> 450. && value <= 2000.) return 2; // slight to moderate restriction
460 else if (value> 2000.) return 3; // severe restriction
461 else return 0;
462//FC
463case 2:
464 if(flagTH==0){
465 if (value == 0.0) return 1; //drinking water
466 else if (value> 0.0 && value <= 200.0) return 2; //recreation primary contact - irrigation crops eaten raw
467 else if (value> 200.0 && value <= 1000.0) return 3; //irrigation
468 else if (value> 1000.0 && value <= 2000.0) return 4; //maximal tolerable class
469 else if (value>2000.0) return 5; //to much
470 else return 0;
471
472 }else if(flagTH==1){ //European bathing directive -> CAUTION: the classes of the directive include more parameters than FC!!!
473 if (value>= 0.0 && value <= 100.0) return 1; //CG - compliant with the mandatory and the guide values of the Directive
474 else if (value> 100.0 && value <= 2000.0) return 2; //CI - compliant with the mandatory values of the Directive
475 else if (value>2000.0) return 3; //NC - not compliant with the mandatory values of the Directive
476
477 }else
478 cout<<"flag: "<<flagTH<<endl;
479 cerr << "ERROR: Wrong flag" << endl;
480 cerr << "usage: only general (0) and European bathing directive thresholds (1) can be used"; exit(1);
481 return NODATA;
482
483
484
485//TN
486case 3: if (value> 0. && value <= 1.) return 1;
487 else if (value> 1. && value <= 1.5) return 2;
488 else if (value> 1.5 && value <= 3.0) return 3;
489 else if (value> 3.0 && value <= 6.0) return 4;
490 else if (value> 6.0 && value <= 12.0) return 5;
491 else if (value> 12.0 && value <= 24.0) return 6;
492 else if (value> 24.0) return 7;
493 else return 0;
494
495//TP
496case 4: if (value> 0. && value <= 0.05) return 1;
497 else if (value> 0.05 && value <= 0.08) return 2;
498 else if (value> 0.08 && value <= 0.15) return 3;
499 else if (value> 0.15 && value <= 0.3) return 4;
500 else if (value> 0.3 && value <= 0.6) return 5;
501 else if (value> 0.6 && value <= 1.2) return 6;
502 else if (value> 1.2) return 7;
503 else return 0;
504
505default:
506 cerr << "ERROR: Wrong parameter type" << endl;
507 cerr << "usage: 0 (BOD), 1 (salt), 2 (Total coliforms), 3 (TN), 4 (TP)\n so far only 0, 3, 4 are implemented"; exit(1);
508 break;
509
510}
