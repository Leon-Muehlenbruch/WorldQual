328int getNrStations(const int parameter_id, const string years, const options_wqstatClass* options_wqstat_ptr){
329 int number;
330
331 mysqlpp::Query query = con.query();
332 query.reset();
333 query <<"SELECT count(distinct w1.`station_id`) as 'number' FROM wq_general.wq_station w, "
334 <<"wq_general.valid_data_"<<options_wqstat_ptr->continent_abb<<" w1, watergap_unf.gcrc g "
335 <<"WHERE w.`station_id`=w1.`station_id` AND w1.`parameter_id`="<<parameter_id
336 <<" AND year(w1.`date`) in ("<<years<<") AND w1.`watertype`="<<options_wqstat_ptr->watertype
337 <<" AND g.`GLOBALID` = w.`global_id` AND w1.`data_used` is null AND w.`data_used` is null"
338 <<" AND g.`IDVersion`="<< options_wqstat_ptr->IDVersion <<" AND g.`IDReg`= "<< options_wqstat_ptr->IDReg
339 <<" ORDER BY w.`rivername`,w1.`station_id`;";
340
341 cout<<"query stationId: "<<query<<endl;
342 res = query.store();
343 if (res.empty()) {cerr << "river_start_end: error no stations available for this parameter.\n"; }
344
345 number = res.at(0)["number"];
346
347
348 return number;
349}

394int getRiverPath(const int Cell_value, int stat, const int nr_stations, int &end_id, int &count,
395 string &all_station_ids, int (*stations_ds)[3], int (*station_id)[3], int (*arc_glob_id)[2],
396 const options_wqstatClass* options_wqstat_ptr, const continentStruct* continent_ptr){
397 int outflowCell = get_outflowCell(Cell_value, options_wqstat_ptr);
398
399
400 //if the cell is not a sink
401 if(outflowCell!=0){
402 for(int j=0; j<nr_stations;j++){
403 //looks for all stations with the same arc id
404 if(station_id[j][1]==Cell_value){
405 stat++;
406 //end id is set
407 end_id=station_id[j][2];
408 //number of stations are counted
409 count++;
410 //stations on path are stored in a string seperated by $
411 std::stringstream help;
412 help<<all_station_ids<<station_id[j][0]<<"$";
413 all_station_ids=help.str();
414 //station is worked off in ds and help
415 stations_ds[j][1]=1;
416 station_id[j][1]=-9999;
417 }
418 }
419
420 //cout<<"in getRiverPath end_id"<<end_id<<" count: "<<count<<" all_station_ids: "<<all_station_ids<<" stat: "<<stat<<endl;
421
422 //start recursion
423 stat=getRiverPath(outflowCell, stat, nr_stations, end_id, count, all_station_ids, stations_ds, station_id,arc_glob_id, options_wqstat_ptr, continent_ptr);
424 }else {
425
426
427 end_id=arcid2globid(Cell_value, arc_glob_id, continent_ptr);
428
429 }
430 return stat;
431
432}
