408void get_Output(vector<statisticStationsClass>& statistic, int Cell_value, int toCalc
409 , int parameter_id, int parameter_id_load
410 , int parameter_id_discharge, int parameter_id_conductivity
411 , int start, int end, int IDrun, char *MyDatabase, const options_wqstatClass* options_wqstat_ptr)
412{
413 int j;
414 char sql[2000], sql1[2000], sql2[2000]; // variable for SQL-query
415 //cout << "get_Output():\n";
416
417 // monatliche Mittelwerte von Messwerten fuer die Zelle Cell_value
418 j=sprintf(sql1, "SELECT a.`parameter_id`, -9999, a.`cell`, a.`month` AS `date`, a.`flag_uncert`, avg(a.measured_day_avg) AS `measured` ");
419 j+=sprintf(sql1+j, "FROM (SELECT w.`parameter_id`, w.`date`, ws.`global_id`, g.`cell`, max(ifnull(ws.`flag_uncert`,0)) as 'flag_uncert' ");
420 j+=sprintf(sql1+j, ", avg(w.Value) as 'measured_day_avg', DATE_FORMAT(w.`date`,'%%Y%%m') as `month` ");
421 j+=sprintf(sql1+j, "FROM wq_general.valid_data_%s w, wq_general.wq_station ws, watergap_unf.gcrc g ", options_wqstat_ptr->continent_abb);
422 j+=sprintf(sql+j, "WHERE w.`parameter_id` =%d ", parameter_id);
423 j=sprintf(sql2, " AND w.`data_used` is null AND ws.`data_used` is null AND w.`watertype`=%d ", options_wqstat_ptr->watertype);
424 j+=sprintf(sql2+j, "AND ws.station_id=w.station_id AND g.`IDVersion`=%d AND g.`IDReg`=%d ", options_wqstat_ptr->IDVersion, options_wqstat_ptr->IDReg);
425 j+=sprintf(sql2+j, " AND g.`GLOBALID`=ws.`global_id` AND year(w.`date`) BETWEEN %d AND %d AND g.cell = %d", start, end, Cell_value);
426 j+=sprintf(sql2+j, " GROUP BY w.`parameter_id`, w.`date`, g.`cell`) AS a ");
427 j+=sprintf(sql2+j, "GROUP BY a.`parameter_id`, a.`month`, a.`cell`;");
428
429 // Messwerte von parameter_id
430 j=sprintf(sql, "%s\n WHERE w.`parameter_id` =%d \n %s", sql1, parameter_id, sql2);
431
432 query.reset();
433 query << sql;
434 //cout<<"query statistic_instream: "<<query<<endl;
435 mysqlpp::StoreQueryResult res_statistic_instream = query.store();
436 //cout << "##################################\n";
437 //cout << "res_statistic_instream.size():\t"<<res_statistic_instream.size()<<endl;
438 //cout << "##################################\n";
439
440 // Messwerte von parameter_id_conductivity
441 mysqlpp::StoreQueryResult res_statistic_conductivity;
442 if (toCalc==1) {
443 j=sprintf(sql, "%s\n WHERE w.`parameter_id` =%d \n %s", sql1, parameter_id_conductivity, sql2);
444
445 query.reset();
446 query << sql;
447 //cout<<"query statistic_instream: "<<query<<endl;
448 res_statistic_conductivity = query.store();
449 //cout << "##################################\n";
450 //cout << "res_statistic_conductivity.size():\t"<<res_statistic_conductivity.size()<<endl;
451 //cout << "##################################\n";
452 }
453
454 if ((toCalc!=1 && res_statistic_instream.size() == 0)
455 || (toCalc==1 && res_statistic_instream.size() == 0 && res_statistic_conductivity.size() == 0)) return;
456
457 // Minimum und Maximum Datum von den Messwerten, um berechnete Werte in gleichen Zeitraum aber lueckenlos auszugeben
458 int min_date, max_date, min_year, max_year, min_month, max_month, beg_mon, end_mon;
459 if (res_statistic_instream.size()>0) {
460 min_date = res_statistic_instream[0]["date"];
461 max_date = res_statistic_instream[0]["date"];
462 } else {
463 min_date = res_statistic_conductivity[0]["date"];
464 max_date = res_statistic_conductivity[0]["date"];
465 }
466 findMinMaxDate(min_date, max_date, res_statistic_instream);
467 findMinMaxDate(min_date, max_date, res_statistic_conductivity);
468
469 // wenn Zeitraum so anfangen und enden soll wie Messwerte gibt, min_month, max_month so berechnen
470 min_year = min_date/100; min_month=min_date%100;
471 max_year = max_date/100; max_month=max_date%100;
472 // Jahr immer ab Januar bis Dezember
473 min_month = 1; max_month = 12;
474
475 statisticStationsClass statistic_line;
476 statistic_line.setValues();
477
478 for (int year = min_year; year<=max_year; year++) {
479 beg_mon=1; end_mon=12;
480 if (year==min_year) beg_mon=min_month;
481 else if (year==max_year) end_mon=max_month;
482
483 for (int month=beg_mon; month<=end_mon; month++) {
484 j=sprintf(sql, "SELECT c.`date`, c.`con`/1000 as `con1_mg_l`, c.`con2`/1000 as `con2_mg_l` ");
485 j+=sprintf(sql+j, "FROM `%swq_out_%s`.`concentration_%d` c ", MyDatabase, options_wqstat_ptr->continent_abb, IDrun);
486 j+=sprintf(sql+j, "WHERE c.`date`=%d ", year*100+month);
487 j+=sprintf(sql+j, "AND c.cell=%d;", Cell_value);
488
489 query.reset();
490 query << sql;
491 //cout<<"query statistic_stations: "<<query<<endl;
492 mysqlpp::StoreQueryResult res_statistic_stations = query.store();
493
494 for (unsigned int rowNum=0; rowNum<res_statistic_stations.size(); rowNum++) {
495 statistic_line.cell = Cell_value;
496 statistic_line.date = year*100+month;
497 statistic_line.con = res_statistic_stations[rowNum]["con1_mg_l"];
498 statistic_line.con2 = res_statistic_stations[rowNum]["con2_mg_l"];
499 statistic_line.measured = NODATA;
500 statistic_line.flag_uncert = NODATA;
501 statistic_line.measured_conductivity = NODATA;
502 statistic_line.flag_uncert_conductivity = NODATA;
503
504 for(unsigned int num=0; num<res_statistic_instream.size(); num++) {
505 int date = res_statistic_instream[num]["date"];
506 if (date==year*100+month) {
507 statistic_line.measured = res_statistic_instream[num]["measured"];
508 statistic_line.flag_uncert = res_statistic_instream[num]["flag_uncert"];
509 }
510 }
511 for(unsigned int num=0; num<res_statistic_conductivity.size(); num++) {
512 int date = res_statistic_conductivity[num]["date"];
513 if (date==year*100+month) {
514 statistic_line.measured_conductivity = res_statistic_conductivity[num]["measured"];
515 statistic_line.flag_uncert_conductivity = res_statistic_conductivity[num]["flag_uncert"];
516 }
517 }
518 statistic.push_back(statistic_line);
519 }
520
521 } // for(month)
522 } // for(year)
523
524 //cout << "statistic.size: " << statistic.size() << endl;
525 //cout << "##################################\n";
526} // end of get_Output()
