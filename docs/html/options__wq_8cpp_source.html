33 //#############
34 // set default options

46 path_hydro_routing = new char[maxcharlength];
47 loadingType = 0;
48
49 //#############
50 // read OPTIONS.DAT
51 int parent_number=12; // ab diesem Nummer sind Optionen von der Klasse optionsClass
52
53 FILE *file_ptr;
54 file_ptr = fopen(filename_opt, "r");
55 if (file_ptr != NULL) {
56 printf("Reading: %s\n\n", filename_opt);
57 int i = 0, n;
58
59 // so lange options.dat noch nicht beendet und Zeilen von der optionsClass nicht angefangen
60 while (fgets(string, maxcharlength, file_ptr) && i<parent_number-1) {
61 if (0 == strncmp(string, "Value:", 6)) {
62 cout << i+1 << string << endl;
63 i++;
64 n = atoi(&string[7]);
65 if (1 == i) {
66 if (n == 0 || n == 1)
67 riverLength = n;

69 cerr << "OPTIONS ERROR: 1. Option 'river length' darf 0 oder 1 sein!\n";
70 return 1;

72 }
73 if (2 == i) {
74 if (n == 0 || n == 1)
75 routingOrder = n;
76 else {
77 cerr << "OPTIONS ERROR: 2. Option 'routing order' darf 0 oder 1 sein!\n";
78 return 1;
79 }
80 }
81 if (3 == i) {
82 if (n == 0 || n == 1 || n == 2)
83 flowVelocityVariant = n;

85 cerr << "OPTIONS ERROR: 8. Option 'Flie�geschwindigkeit'.\n";
86 cerr << " 0: mit variablen Flie�geschwindigkeit aus der Tabelle oder UNF-Datei als Standart ###\n";

91 }
92 if (4 == i) {
93 if (flowVelocityVariant == 2)
94 ConstVelocity = atof(&string[7]);
95 else ConstVelocity = 0.;
96 }
97 if (5 == i) {
98 if (n == 0 || n == 1 || 2)
99 climate = n;
100 else {
101 cerr << "OPTIONS ERROR: 10. Option 'Klimainput' darf 0 oder 1 sein!\n";
102 cerr << " 0: Klimainput in 0.5 \n";
103 cerr << " 1: Klimainput 5 min\n";
104 cerr << " 2: Klimainput in 0.5 CRU/WATCH\n";
105 return 1;
106 }
107 }
108 if (6 == i) {
109 if (n == 0 || n == 1)
110 isWaterTempFromAir = n;
111 else {
112 cerr << "OPTIONS ERROR: 6. water temperature darf 0 oder 1 sein!\n";
113 return 1;
114 }
115 }
116 if (7 == i) {
117 StringCheck(&string[7]);
118 strcpy(path_watergap_output, &string[7]);
119 }
120 if (8 == i) {
121 StringCheck(&string[7]);
122 strcpy(path_climate, &string[7]);
123 }
124 if (9 == i) {
125 StringCheck(&string[7]);
126 strcpy(path_hydro_input, &string[7]);
127 }
128 if (10 == i) {
129 StringCheck(&string[7]);
130 strcpy(path_hydro_routing, &string[7]);
131 }
132 if (11 == i) {
133 loadingType = n;
134 }
135 } // if(Value)
136 }
137
138 if (read_parentOptions(parent_number, file_ptr)) return 1;
139 fclose(file_ptr);
140 } else {
141 printf("\nError while opening file '%s'.\n", filename_opt);

148
149 // Ausgabe worldqual-Optionen
150 if (riverLength)
151 cout << "calculate river length" <<endl;
152 else
153 cout << "no calculate river length" <<endl;
154 if (routingOrder)
155 cout << "calculate routing order" <<endl;
156 else
157 cout << "no calculate routing order" <<endl;
158
159 switch (flowVelocityVariant) {
160 case 0:
161 if (!InputType) cout << "mit variablen Fliessgeschwindigkeit aus der Tabelle\n";
162 else cout << "mit variablen Fliessgeschwindigkeit aus UNF-Dateien\n";
163 break;
164 case 1: cout << "mit variablen Fliessgeschwindigkeit (berechnet im Programm)\n"; break;
165 case 2: cout << "mit konstanten Fliessgeschwindigkeit = "<< ConstVelocity <<'\n'; break;
166 }
167
168 switch (climate) {
169 case 0: cout << "Climate Input: 0.5°\n"; break;
170 case 1: cout << "Climate Input: 5min\n"; break;
171 case 2: cout << "Climate Input: 0.5° CRU/WATCH\n"; break;
172 }
173 if (isWaterTempFromAir)
174 cout << "Water temperature is from DB, IDTemp from wq_general._runlist" <<endl;
175 else {
176 cout << "Water temperature calculated from air temperature" <<endl;
177 if (climate==0) cout << " wird G_ELEV_MEAN_CLM.UNF0 benötigt." << endl;
178 else if (climate==2) cout << " wird G_ELEV_MEAN_WATCH.UNF0 benötigt." << endl;
179 }
180 cout << "path to watergap_output : "<< path_watergap_output << "\n";
181 cout << "path to climate : "<< path_climate << "\n";
182 cout << "path to hydro_input : "<< path_hydro_input << "\n";
183 cout << "path to path_hydro_routing: "<< path_hydro_routing << "\n";
184
185 switch(loadingType){
186 case 0: cout << "Loading is in the Table load\n"; break;
187 case 1: cout << "Loading is in the Table load_[IDScen]_[parameter_id]\n"; break;
188 case 2: cout << "Loading read from wq_load OUTPUT (Tables calc_cell_monthly_load_[type])\n"; break;
189 }
190 optionsClass::printOptions(); // Ausgabe allgemeinen Optionen, continent_abb wird hier gef�llt
191
192 return 0;
