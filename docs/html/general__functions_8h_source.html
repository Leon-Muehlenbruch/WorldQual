550template <class T> int readInputFile(char *input_file, int type_size, int valuesPerCell, long number_of_cells
551 , T *G_array, int *gcrc, const optionsClass* options_ptr, const continentStruct* continent_ptr, float nodata=(T)NODATA)
552{
553 cout <<"readInputFile for template; sizeof = " <<sizeof(T)<< "\n";
554 T nodata_value;
555 if(sizeof(nodata_value)==1) nodata_value=0;
556 else nodata_value=nodata;
557 ifstream ifile(input_file);
558 if (!ifile) { cerr << "Can't open input file "<< input_file<<endl; return 1;}
559 T* G_tmp_array = new T[number_of_cells*valuesPerCell];
560 long bytesRead;
561 ifile.read((char*)G_tmp_array, number_of_cells*valuesPerCell*type_size);
562 bytesRead = ifile.gcount();
563 cout << "file "<< input_file << ' '<< bytesRead <<" bytesRead\n"
564 <<"number_of_cells: " <<number_of_cells<<" valuesPerCell: "<<valuesPerCell
565 <<" type_size: "<<type_size<<' ';
566 if (type_size==1 ) cout <<"G_tmp_array "<< (int)G_tmp_array[0]<<endl;
567 else cout <<"G_tmp_array "<<G_tmp_array[5157]<<endl;
568 ifile.close();
569 if (bytesRead!=number_of_cells*valuesPerCell*type_size) {
570 cerr << "Input file size mismatch "<< input_file<<" bytesRead: "<<bytesRead<<" number_of_cells*valuesPerCell*type_size: "<<number_of_cells*valuesPerCell*type_size<<endl;
571 return 1;
572 }
573 // Aendern der Reihenfolge der Bytes (SimMemorySwap)
574 #ifdef BYTESWAP_NECESSARY
575 SimMemorySwap((char*) G_tmp_array,
576 (char*) G_tmp_array,
577 type_size,
578 number_of_cells*valuesPerCell*type_size);
579 #endif
580 cout << "file "<< input_file << ' '<< bytesRead <<" bytesRead\n"
581 <<"number_of_cells: " <<number_of_cells<<" valuesPerCell: "<<valuesPerCell
582 <<" type_size: "<<type_size<<' ';
583 if (type_size==1 ) cout << (int)G_tmp_array[0]<<endl;
584 else cout << G_tmp_array[5157]<<endl;
585
586 bool order;
587 //WaterGAP2 has rows and columns in an different order than WG3
588 //This only applies to UNF files in the row format
589 if(strcmp(options_ptr->continent_abb,"wg2")==0 && (number_of_cells==continent_ptr->nrows)){
590 order=false; //cout<<"order false"<<endl;
591 }else{
592 order=true; //cout<<"order true"<<endl;
593 }
594
595 // gcrc -> arcid
596 int index;
597 for (int cell=0; cell<number_of_cells; cell++) {
598 for (int month=0; month<valuesPerCell; month++) {
599 if (number_of_cells == continent_ptr->ng || number_of_cells== continent_ptr->ng_land)
600 index = (gcrc[cell]-1)*valuesPerCell+month;
601 else index = cell*valuesPerCell+month;
602 // gcrc[cell] == arcid
603 //G_array[index] = G_tmp_array[cell*valuesPerCell+month];
604 if (order) G_array[index] = G_tmp_array[cell*valuesPerCell+month];
605 else G_array[index] = G_tmp_array[month*number_of_cells+cell];
606 if (G_array[index]!=G_array[index]) {
607 cerr << "file\t"<<input_file<<"\tcell\t"<<cell<<"\tgcrc\t"<<gcrc[cell]
608 <<"\tmonth\t"<<month+1<<'\t'<<G_array[index]<<"\tvalue ="<<NODATA<<endl;
609 G_array[index]=nodata_value;
610 }
611 }
612 }// for(cell)
613
614
615 //cout<<"before delete "<<endl;
616 delete[] G_tmp_array; G_tmp_array = NULL;
617 //cout<<"before return"<<endl;
618 return 0;
619}
