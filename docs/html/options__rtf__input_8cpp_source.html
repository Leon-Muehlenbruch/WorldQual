32short options_rtfinput::init(int option_c, char* option_v[])
33{
34 char string[maxcharlength];
35
36 //#############
37 // set default options

39 optionsClass::init(option_c, option_v); // initiiren allgemeine Optionen
40 project_id = 4;

42 input_dir_griddi = new char[maxcharlength];

47 strcpy(input_dir_corrfactor,".");
48 IDInTableName = 0;

50 //#############
51 // read OPTIONS.DAT
52 int parent_number=7; // ab diesem Nummer sind Optionen von der Klasse optionsClass
53
54 FILE *file_ptr;
55 file_ptr = fopen(filename_opt, "r");
56 if (file_ptr != NULL) {
57 printf("Reading: %s\n\n", filename_opt);
58 int i = 0, n;
59
60 while (fgets(string, maxcharlength, file_ptr) && i<parent_number-1) {
61 if (0 == strncmp(string, "Value:", 6)) {
62 i++;
63 n = atoi(&string[7]);
64 if (1 == i) {
65 project_id = n;
66 }
67 if (2 == i) {
68 IDScen = n;
69 }
70 if (3 == i) {
71 StringCheck(&string[7]);
72 strcpy(input_dir_griddi, &string[7]);
73 }
74 if (4 == i) {
75 StringCheck(&string[7]);
76 strcpy(input_dir_gwsws, &string[7]);
77 }
78 if (5 == i) {
79 StringCheck(&string[7]);
80 strcpy(input_dir_corrfactor, &string[7]);
81 }
82 if (6 == i) {
83 IDInTableName = n;
84 }
85 } // if(Value)
86 }
87 if (read_parentOptions(parent_number, file_ptr)) return 1;
88 fclose(file_ptr);
89 } else {
90 printf("\nError while opening file '%s'.\n", filename_opt);
91 //printf("Using default options.\n");
92 return 1;
93 }
94
95 //#############
96 // Options Ausgabe
97
98 // Ausgabe scen-Optionen
99 cout << "project_id: \t" << project_id<< endl;
100 cout << "scenario IDScen: \t" << IDScen << endl;
101 cout <<"input_dir_griddi : "<< input_dir_griddi<<endl<<endl;
102 cout <<"input_dir_gwsws : "<< input_dir_gwsws<<endl<<endl;
103 cout <<"input_dir_corrfactor: "<< input_dir_corrfactor<<endl<<endl;
104
105 optionsClass::printOptions(); // Ausgabe allgemeinen Optionen, continent_abb wird hier gef�llt
