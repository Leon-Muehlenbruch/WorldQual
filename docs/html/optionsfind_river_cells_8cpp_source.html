45short options_findRiverCellsClass::init(int option_c, char* option_v[])
46{
47 char string[maxcharlength];
48
49 //#############
50 // set default options
51 // which will be used, when OPTIONS.DAT is not found
52 optionsClass::init(option_c, option_v); // initiiren allgemeine Optionen

55
56 startGlobal = NODATA;

60
61 //#############
62 // wenn in der Befehlszeile start- und/oder end-GlobalID eingegeben, lesen
63 char option[5]; // to process options, e.g. "-d", "-t", "-o", "-r", "-s"
64 char optionValue[maxcharlength]; // option value (filenames)
65
66
67 for (int num=0; num<option_c; num++) {
68 cout << num << endl;
69 separateCommandLineOption(option_v[num], option, optionValue);
70 cout << "---\n";

72 startGlobal = atoi(optionValue);
73 useInputFile = false;
74 }
75 else if (!strcmp(option,"-e")) {
76 endGlobal = atoi(optionValue);
77 useInputFile = false;
78 }
79 } // for(i)
80 cout << "options done\n";
81
82 //#############
83 // read OPTIONS.DAT
84 int parent_number=2; // ab diesem Nummer sind Optionen von der Klasse optionsClass
85
86 FILE *file_ptr;
87 file_ptr = fopen(filename_opt, "r");
88 if (file_ptr != NULL) {
89 printf("Reading: %s\n\n", filename_opt);
90 int i = 0;
91
92 while (fgets(string, maxcharlength, file_ptr) && i<parent_number-1) {
93 if (0 == strncmp(string, "Value:", 6)) {
94 i++;
95 StringCheck(&string[7]);
96 strcpy(inputfile, &string[7]);
97 } // if(Value)
98 }
99 if (read_parentOptions(parent_number, file_ptr)) return 1;
100 fclose(file_ptr);
101 } else {
102 printf("\nError while opening file '%s'.\n", filename_opt);
