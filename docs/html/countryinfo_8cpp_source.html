48void countryInfoClass::init(const char *input_dir,
49 const unsigned short maxLengthOfCountryName)
50{
51 // in the default case only the first 15 characters
52 // of the country names are used.
53
54 char filename[250];
55 sprintf(filename,"%s/COUNTRY_NAMES.DAT",input_dir);
56 ifstream countryFile(filename);
57 if (!countryFile){
58 cerr << "Error while opening file " << filename << "." << endl;
59 exit(1);
60 }
61 else {
62 char *name;
63 char string[250];
64 short length, j , i=0;
65
66 // read commentlines indicated by #
67 while (!countryFile.eof() && countryFile.peek()=='#') {
68 countryFile.getline(string,sizeof(string));
69 }
70 // set data type to 'decimal'
71 // otherwise some compiler assume that numbers with leading 'zeros'
72 // are octal
73 cin.setf(ios::dec, ios::basefield);
74
75 while (countryFile >> j >> ws){
76 // read iso number and following whitespaces
77
78 countryIsoNumber.push_back(j);
79
80 // read country name.
81 // country names might include space characters
82 // therefore 'getline' is used.
83 countryFile.getline(string,sizeof(string));
84 length = strlen(string);
85
86 // allocate memory and copy string with the desired length into it
87 if (length >= maxLengthOfCountryName) {
88 name = new char[maxLengthOfCountryName+1];
89 length = maxLengthOfCountryName;
90 strncpy(name, string, length);
91 }
92 else {
93 name = new char[length+1];
94 strncpy(name, string, length);
95 name[length]='\0';
96 }
97 countryName.push_back(name);
98 i++;
99 }
100 countryFile.close();
101
102 numberOfCountries = i;
103
104 // avoid unused memory
105 countryIsoNumber.resize(numberOfCountries);
106 countryName.resize(numberOfCountries);
107
108 for (int i=0; i<numberOfCountries; i++)
109 cout << i << " "<<countryIsoNumber[i]<<" "<<countryName[i]<<endl;
110
111 }
112}

131{

133 = std::find(countryIsoNumber.begin(), countryIsoNumber.end(), IsoNumber);
